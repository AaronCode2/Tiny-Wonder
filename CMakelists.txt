cmake_minimum_required(VERSION 3.10...3.50)
project(Tiny_Wonder)

include(FetchContent)

set(RAYLIB_VERSION 5.5)

FetchContent_Declare(
    raylib
    URL https://github.com/raysan5/raylib/archive/refs/tags/${RAYLIB_VERSION}.tar.gz
    FIND_PACKAGE_ARGS ${RAYLIB_VERSION} EXACT
)

set(BUILD_EXAMPLES OFF CACHE INTERNAL "")
set(BUILD_GAMES OFF CACHE BOOL "" FORCE)

FetchContent_MakeAvailable(raylib)

set(
SOURCE_FILES

    Src/Main.cpp
    Src/Game.cpp
    Src/App.cpp
    Src/Utils.cpp
    Src/GlobalVars.cpp
    Src/Manager_Classes/Elements.cpp 
    Src/Manager_Classes/UI.cpp
    Src/Manager_Classes/World.cpp
    Src/Manager_Classes/TileManager.cpp
    Src/Manager_Classes/NeighborTypes.cpp
    Src/Manager_Classes/DataIO.cpp
    Src/Entity_Classes/Tile.cpp
    Src/Entity_Classes/Sprite.cpp
    Src/Entity_Classes/Player.cpp
    Src/Entity_Classes/Plant.cpp
    Src/Entity_Classes/Dealer.cpp
)

add_executable(Tiny_Wonder ${SOURCE_FILES})
target_link_libraries(Tiny_Wonder raylib)

if(WIN32)

    target_sources(Tiny_Wonder PRIVATE Assets/Windows/app.rc)
    target_link_options(Tiny_Wonder PRIVATE $<$<CONFIG:Release>:-mwindows>)
elseif(APPLE)

    target_link_options(Tiny_Wonder PRIVATE $<$<CONFIG:Release>:-MACOSX_BUNDLE>)
endif()

target_compile_definitions(Tiny_Wonder PUBLIC $<$<CONFIG:Debug>:DEBUG_ENABLED>)

if("${PLATFORM}" STREQUAL "Web")

    message(FATAL_ERROR "Do not build in this project in the Web as it doesn't work!")
endif()

if(APPLE)

    target_link_libraries(${PROJECT_NAME} "-framework IOKit")
    target_link_libraries(${PROJECT_NAME} "-framework Cocoa")
    target_link_libraries(${PROJECT_NAME} "-framework OpenGL")
endif()
